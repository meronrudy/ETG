44/// Interval random variable helper: draws s^2 values from two event distributions.
45pub fn interval_sample<L, G, S1, S2>(geom: &G, c: Scalar, e1: &mut S1, e2: &mut S2) -> Scalar
46where
47    G: MetricSpace<L>,
48    S1: Sampler<Event<L>>,
49    S2: Sampler<Event<L>>,
50{
51    let a = e1.sample();
52    let b = e2.sample();
53    let dt = b.t - a.t;
54    let dx = geom.dist(&a.l, &b.l);
55    s2(c, dt, dx)
56}

58/// Monte-Carlo estimator for P(E1 â†’ E2) using etg-core's causally_allows.
59pub fn causal_prob<L, G, S1, S2>(geom: &G, c: Scalar, e1: &mut S1, e2: &mut S2, samples: usize) -> Scalar
60where
61    G: MetricSpace<L>,
62    S1: Sampler<Event<L>>,
63    S2: Sampler<Event<L>>,
64{
65    let mut ok = 0usize;
66    for _ in 0..samples {
67        let a = e1.sample();
68        let b = e2.sample();
69        if causally_allows(geom, c, &a, &b) {
70            ok += 1;
71        }
72    }
73    (ok as Scalar) / (samples as Scalar)
74}

76/// Distributional invariance check utility:
77/// Empirically verify mean s^2 difference before/after a fixed numeric boost is ~0.
78pub fn empirical_s2_invariance<L, G, S1, S2>(
79    geom: &G,
80    c: Scalar,
81    v: Scalar,
82    e1: &mut S1,
83    e2: &mut S2,
84    samples: usize,
85) -> Scalar
86where
87    G: MetricSpace<L>,
88    S1: Sampler<Event<L>>,
89    S2: Sampler<Event<L>>,
90{
91    let mut acc = 0.0;
92    for _ in 0..samples {
93        let a = e1.sample();
94        let b = e2.sample();
95        let dt = b.t - a.t;
96        let dx = geom.dist(&a.l, &b.l);
97        let (dtp, dxp) = boost(c, v, dt, dx);
98        let d = s2(c, dtp, dxp) - s2(c, dt, dx);
99        acc += d;
100    }
101    acc / (samples as Scalar)
102}

114/// Seeded event distribution: owns a reproducible RNG and a generator closure.
115///
116/// Example (doctest):
117/// ```
118/// use etg_core::{Event, Euclid1D};
119/// use etg_prob::{SeededEventDist, causal_prob_with_ci, EventDist};
120/// use rand::rngs::StdRng;
121/// use rand::SeedableRng;
122/// use rand::Rng;
123///
124/// let geom = Euclid1D;
125/// let c = 1.0;
126///
127/// // E1 fixed
128/// let mut e1 = EventDist::new(|| Event{ l: 0.0_f64, t: 0.0 });
129///
130/// // E2 random but reproducible: t in {1,3} with equal probability at fixed l
131/// let mut e2 = SeededEventDist::new_seeded(42, |rng: &mut StdRng| {
132///     let t = if rng.gen_bool(0.5) { 3.0 } else { 1.0 };
133///     Event { l: 2.0_f64, t }
134/// });
135///
136/// let (p, lo, hi) = causal_prob_with_ci(&geom, c, &mut e1, &mut e2, 20_000, 1.96);
137/// assert!(p >= lo && p <= hi);
138/// ```
139pub struct SeededEventDist<L, F: FnMut(&mut StdRng) -> Event<L>> {
140    rng: StdRng,
141    f: F,
142}
143
144impl<L, F: FnMut(&mut StdRng) -> Event<L>> SeededEventDist<L, F> {
145    pub fn new_seeded(seed: u64, f: F) -> Self {
146        Self { rng: StdRng::seed_from_u64(seed), f }
147    }
148    #[inline]
149    pub fn sample(&mut self) -> Event<L> { (self.f)(&mut self.rng) }
150}

157/// Monte-Carlo estimator with Wald-style confidence interval.
158///
159/// Returns (p_hat, ci_lo, ci_hi) using half-width z * sqrt(p_hat * (1 - p_hat) / n).
160///
161/// Note: For small-sample or near-boundary cases, consider Wilson/Agresti-Coull.
162/// This simple Wald interval is adequate for large n and non-degenerate probabilities.
163///
164/// Example (doctest):
165/// ```
166/// use etg_core::{Event, Euclid1D};
167/// use etg_prob::{EventDist, causal_prob_with_ci};
168///
169/// let geom = Euclid1D;
170/// let c = 1.0;
171/// let mut e1 = EventDist::new(|| Event { l: 0.0_f64, t: 0.0 });
172/// let mut e2 = EventDist::new(|| Event { l: 0.0_f64, t: 1.0 }); // always causal
173///
174/// let (p, lo, hi) = causal_prob_with_ci(&geom, c, &mut e1, &mut e2, 5_000, 1.96);
175/// assert!(p > 0.98 && lo > 0.95 && hi <= 1.0);
176/// ```
177pub fn causal_prob_with_ci<L, G, S1, S2>(
178    geom: &G,
179    c: Scalar,
180    e1: &mut S1,
181    e2: &mut S2,
182    samples: usize,
183    z: Scalar,
184) -> (Scalar, Scalar, Scalar)
185where
186    G: MetricSpace<L>,
187    S1: Sampler<Event<L>>,
188    S2: Sampler<Event<L>>,
189{
190    let p = causal_prob(geom, c, e1, e2, samples);
191    let var = if samples > 1 { p * (1.0 - p) / (samples as Scalar) } else { 0.0 };
192    let half = z * var.sqrt();
193    let lo = (p - half).max(0.0);
194    let hi = (p + half).min(1.0);
195    (p, lo, hi)
196}

198/// Adaptive Monte-Carlo for causal probability with target CI half-width eps.
199///
200/// - z: z-score (e.g., 1.96 for ~95%)
201/// - eps: desired half-width tolerance
202/// - max_samples: ceiling to prevent runaway loops
203///
204/// Returns (p_hat, n_used). If early stopping fails, uses max_samples.
205///
206/// Example (doctest):
207/// ```
208/// use etg_core::{Event, Euclid1D};
209/// use etg_prob::{EventDist, causal_prob_adaptive};
210///
211/// let geom = Euclid1D;
212/// let c = 1.0;
213/// let mut e1 = EventDist::new(|| Event { l: 0.0_f64, t: 0.0 });
214/// let mut e2 = EventDist::new(|| Event { l: 2.0_f64, t: 3.0 }); // always causal at c=1
215///
216/// let (p, n) = causal_prob_adaptive(&geom, c, &mut e1, &mut e2, 100_000, 1e-3, 1.96);
217/// assert!(p > 0.999);
218/// assert!(n >= 100);
219/// ```
220pub fn causal_prob_adaptive<L, G, S1, S2>(
221    geom: &G,
222    c: Scalar,
223    e1: &mut S1,
224    e2: &mut S2,
225    max_samples: usize,
226    eps: Scalar,
227    z: Scalar,
228) -> (Scalar, usize)
229where
230    G: MetricSpace<L>,
231    S1: Sampler<Event<L>>,
232    S2: Sampler<Event<L>>,
233{
234    let mut ok = 0usize;
235    let mut n = 0usize;
236
237    while n < max_samples {
238        n += 1;
239        let a = e1.sample();
240        let b = e2.sample();
241        if causally_allows(geom, c, &a, &b) {
242            ok += 1;
243        }
244
245        // After a modest warmup, check CI half-width
246        if n >= 100 {
247            let p = ok as Scalar / n as Scalar;
248            let se = (p * (1.0 - p) / (n as Scalar)).sqrt();
249            if z * se <= eps && p.is_finite() {
250                return (p, n);
251            }
252        }
253    }
254
255    let p = ok as Scalar / n as Scalar;
256    (p, n)
257}

259/// Compute mean and (unbiased) variance of s^2 deltas under a fixed boost using Welford's method.
260///
261/// Returns (mean_delta, var_delta).
262pub fn empirical_s2_delta_stats<L, G, F1, F2>(
263    geom: &G,
264    c: Scalar,
265    v: Scalar,
266    e1: &mut EventDist<L, F1>,
267    e2: &mut EventDist<L, F2>,
268    samples: usize,
269) -> (Scalar, Scalar)
270where
271    G: MetricSpace<L>,
272    F1: FnMut() -> Event<L>,
273    F2: FnMut() -> Event<L>,
274{
275    let mut n: usize = 0;
276    let mut mean: Scalar = 0.0;
277    let mut m2: Scalar = 0.0;
278
279    for _ in 0..samples {
280        let a = e1.sample();
281        let b = e2.sample();
282        let dt = b.t - a.t;
283        let dx = geom.dist(&a.l, &b.l);
284        let (dtp, dxp) = boost(c, v, dt, dx);
285        let d = s2(c, dtp, dxp) - s2(c, dt, dx);
286
287        n += 1;
288        let delta = d - mean;
289        mean += delta / (n as Scalar);
290        let delta2 = d - mean;
291        m2 += delta * delta2;
292    }
293
294    if n > 1 {
295        (mean, m2 / ((n as Scalar) - 1.0))
296    } else {
297        (mean, 0.0)
298    }
