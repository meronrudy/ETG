11/// A minimal evolution stepper interface.
12pub trait Evolver<G> {
13    fn step(&mut self, g: &mut G, dt: Scalar);
14}

16/// A simple scalar field over a 1D lattice.
17#[derive(Clone, Debug)]
18pub struct Field1D {
19    pub data: Vec<Scalar>,
20}
21impl Field1D {
22    pub fn new(data: Vec<Scalar>) -> Self { Self { data } }
23    #[inline]
24    pub fn len(&self) -> usize { self.data.len() }
25    #[inline]
26    pub fn is_empty(&self) -> bool { self.data.is_empty() }
27    #[inline]
28    pub fn sum(&self) -> Scalar { self.data.iter().copied().sum() }
29}

31/// Check conservation of total sum between two fields within a relative tolerance.
32///
33/// Returns true iff:
34/// |sum(next) - sum(prev)| â‰¤ tol * max(1.0, |sum(prev)|)
35#[inline]
36pub fn conserved_sum(prev: &Field1D, next: &Field1D, tol: Scalar) -> bool {
37    let sp = prev.sum();
38    let sn = next.sum();
39    let diff = (sn - sp).abs();
40    let denom = 1.0_f64.max(sp.abs());
41    diff <= tol * denom
42}

44/**/
45/// A conservative explicit stepper: zero-flux boundaries, symmetric stencil.
46///
47/// new\[i] = old\[i] + alpha*dt*(old\[i-1] - 2*old\[i] + old\[i+1])
48///
49/// Zero-flux boundaries are implemented by mirroring the boundary values:
50/// left = old\[0] for i=0 and right = old\[n-1] for i=n-1. For a uniform field,
51/// the discrete Laplacian vanishes everywhere, preserving total "mass" exactly
52/// up to floating-point rounding.
53#[derive(Clone, Copy, Debug)]
54pub struct DiffuseEvolver {
55    pub alpha: Scalar,
56}
57impl Evolver<Field1D> for DiffuseEvolver {
58    fn step(&mut self, f: &mut Field1D, dt: Scalar) {
59        let n = f.len();
60        if n == 0 { return; }
61        let a = self.alpha * dt;
62        let old = f.data.clone();
63        for i in 0..n {
64            let left  = if i == 0 { old[0] } else { old[i - 1] };
65            let mid   = old[i];
66            let right = if i + 1 == n { old[n - 1] } else { old[i + 1] };
67            f.data[i] = mid + a * (left - 2.0 * mid + right);
68        }
69        // This discretization is conservative under zero-flux boundaries;
70        // numerical drift is bounded by floating precision.
71    }
72}
