17pub type Scalar = f64;
18
19/// A metric provider for location type `L`.
20pub trait MetricSpace<L> {
21    fn dist(&self, a: &L, b: &L) -> Scalar;
22}
23
24/// An event consists of a location and an absolute time (Scalar).
25#[derive(Clone, Debug)]
26pub struct Event<L> {
27    pub l: L,
28    pub t: Scalar,
29}
30
31/// Minkowski-like quadratic form s^2 = c^2 dt^2 - dx^2 over number differences.
32#[inline]
33pub fn s2(c: Scalar, dt: Scalar, dx: Scalar) -> Scalar {
34    c * c * dt * dt - dx * dx
35}
36
37/// Interval between two events via a geometry provider and invariant speed c.
38#[inline]
39pub fn interval<L, G: MetricSpace<L>>(geom: &G, c: Scalar, e1: &Event<L>, e2: &Event<L>) -> Scalar {
40    let dt = e2.t - e1.t;
41    let dx = geom.dist(&e1.l, &e2.l);
42    s2(c, dt, dx)
43}
44
45/// Causality predicate: dt >= dx / c
46#[inline]
47pub fn causally_allows<L, G: MetricSpace<L>>(geom: &G, c: Scalar, e1: &Event<L>, e2: &Event<L>) -> bool {
48    let dt = e2.t - e1.t;
49    let dx = geom.dist(&e1.l, &e2.l);
50    dt >= dx / c
51}
52
53/// 1D Lorentz-like boost over differences (reference implementation).
54/// Precondition: |v| < c, c > 0.
55#[inline]
56pub fn boost(c: Scalar, v: Scalar, dt: Scalar, dx: Scalar) -> (Scalar, Scalar) {
57    let gamma = 1.0 / (1.0 - (v * v) / (c * c)).sqrt();
58    let dtp = gamma * (dt - (v * dx) / (c * c));
59    let dxp = gamma * (dx - v * dt);
60    (dtp, dxp)
61}

63/// A simple Euclidean metric in 1D for examples/tests.
64pub struct Euclid1D;
65
66impl MetricSpace<f64> for Euclid1D {
67    #[inline]
68    fn dist(&self, a: &f64, b: &f64) -> Scalar {
69        (a - b).abs()
70    }
71}

78/// Admissibility interface for frames/transforms.

80pub trait LorentzLike {
